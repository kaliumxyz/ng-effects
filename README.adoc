== Angular Effects

Reactive extensions for Angular.

[source,typescript]
----
@Component({
    selector: "app-root",
    template: `
        <div>Count: {{count}}</div>
    `,
    providers: [Effects],
})
export class AppComponent {
    @Input()
    count: number = 0

    @Effect("count")
    incrementCount(state: State<AppState>) {
        return state.count.pipe(
            take(1),
            increment(1),
            repeatInterval(1000)
        )
    }

    constructor(connect: Connect) {
        connect(this)
    }
}
----
